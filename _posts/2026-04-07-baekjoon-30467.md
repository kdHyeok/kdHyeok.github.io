### 출력 

 <p>사람들이 최대한 많은 소원을 빌도록 날려 보낼 등불들을 골랐을 때, 소원을 비는 횟수를 출력한다.</p>

### 코드 (C++)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 목적/이유: 세그먼트 트리의 특정 위치(리프 노드)의 값을 변경(추가/제거)하고 구간 합을 갱신하기 위함.
// 사용법: tree 배열, 현재 노드 1, 시작 범위 1, 끝 범위 max_val, 변경할 압축된 속도 인덱스, 증감값(+1 또는 -1)을 인자로 전달하여 호출한다.
void update(vector<int>& tree, int node, int start, int end, int index, int diff) {
    if (index < start || index > end) return; // 타겟 인덱스가 범위를 벗어나면 종료
    
    tree[node] += diff; // 현재 노드에 증감치 반영
    
    if (start != end) { // 리프 노드가 아니라면 자식 노드들로 재귀 탐색
        int mid = (start + end) / 2;
        update(tree, node * 2, start, mid, index, diff);
        update(tree, node * 2 + 1, mid + 1, end, index, diff);
    }
}

// 목적/이유: 특정 속도 구간(left ~ right)에 존재하는 등불의 개수를 구하기 위함.
// 사용법: 구간 합을 구할 tree 배열, 시작 노드 1, 시작 범위 1, 끝 범위 max_val, 구하고자 하는 범위 left와 right를 전달하여 호출한다.
int query(vector<int>& tree, int node, int start, int end, int left, int right) {
    if (left > end || right < start) return 0; // 목표 구간과 전혀 겹치지 않으면 0 반환
    if (left <= start && end <= right) return tree[node]; // 완전히 속하면 노드 값 반환
    
    int mid = (start + end) / 2; // 일부만 겹치면 분할 정복으로 합산
    return query(tree, node * 2, start, mid, left, right) + 
           query(tree, node * 2 + 1, mid + 1, end, left, right);
}

void solve() {
    // N: 전체 등불의 개수, S: 연속으로 고를 등불의 개수 (윈도우 크기)
    int N, S;
    cin >> N >> S;
    
    // A: 각 등불의 원래 상승 속도를 저장하는 배열
    // sorted_A: 좌표 압축을 위해 중복을 제거하고 정렬할 배열
    vector<int> A(N);
    vector<int> sorted_A(N);
    
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
        sorted_A[i] = A[i];
    }
    
    // [좌표 압축] 
    // 목적/이유: 속도의 범위가 10^9까지이므로, 배열 인덱스로 쓰기 위해 1부터 N사이의 값으로 크기를 줄인다.
    // 사용법: 정렬 후 unique로 중복 제거, 이후 lower_bound를 통해 각 속도의 순위를 찾는다.
    sort(sorted_A.begin(), sorted_A.end());
    sorted_A.erase(unique(sorted_A.begin(), sorted_A.end()), sorted_A.end());
    
    // max_val: 압축된 속도 값의 최대 종류 개수. 트리의 크기를 결정하는 기준이 된다.
    int max_val = sorted_A.size();
    
    // compressed_A: A의 원소들을 1-based의 압축된 인덱스로 변환해 저장하는 배열
    vector<int> compressed_A(N);
    for (int i = 0; i < N; ++i) {
        compressed_A[i] = lower_bound(sorted_A.begin(), sorted_A.end(), A[i]) - sorted_A.begin() + 1;
    }
    
    // tree: 구간 합 세그먼트 트리. 크기는 넉넉히 max_val * 4로 잡는다.
    vector<int> tree(max_val * 4, 0);
    
    // current_wishes: 현재 윈도우 내에서 사람들이 비는 소원의 수(증가하는 쌍의 수). 
    // max_wishes: 모든 윈도우를 탐색하며 갱신할 최대 소원 수. 
    // 이유: 최대 S(100,000)개의 조합에서 쌍의 수는 약 50억 개가 될 수 있으므로 반드시 long long을 사용해야 오버플로우를 막을 수 있다.
    long long current_wishes = 0;
    long long max_wishes = 0;
    
    // [초기 윈도우 설정 (0 ~ S-1)]
    // 목적/이유: 처음 S개의 등불에 대한 소원 수를 구한다.
    for (int i = 0; i < S; ++i) {
        int val = compressed_A[i];
        // 1. 나보다 먼저 들어온 등불 중 속도가 더 느린 것(< val)의 개수를 누적
        current_wishes += query(tree, 1, 1, max_val, 1, val - 1);
        // 2. 내 속도를 트리에 추가
        update(tree, 1, 1, max_val, val, 1);
    }
    
    max_wishes = current_wishes;
    
    // [슬라이딩 윈도우 탐색]
    // 목적/이유: 윈도우를 한 칸씩 우측으로 밀면서 변화하는 소원 수를 O(log N)만에 갱신한다.
    for (int i = S; i < N; ++i) {
        // out_val: 윈도우에서 빠져나갈 맨 앞 원소의 속도
        int out_val = compressed_A[i - S];
        // 1. 빠져나가는 원소가 기여했던 소원 수 제거: 
        // 빠져나가는 애는 그룹의 맨 앞에 있었으므로, 얘보다 속도가 빨랐던 애들(> out_val)과 쌍을 이루었었다. 이를 빼준다.
        current_wishes -= query(tree, 1, 1, max_val, out_val + 1, max_val);
        // 트리에 반영 취소
        update(tree, 1, 1, max_val, out_val, -1);
        
        // in_val: 윈도우에 새로 들어오는 맨 뒤 원소의 속도
        int in_val = compressed_A[i];
        // 2. 새로 들어오는 원소가 기여하는 소원 수 추가:
        // 새로 들어오는 애는 그룹의 맨 뒤에 있으므로, 얘보다 속도가 느렸던 애들(< in_val)과 쌍을 이룬다. 이를 더해준다.
        current_wishes += query(tree, 1, 1, max_val, 1, in_val - 1);
        // 트리에 새 원소 반영
        update(tree, 1, 1, max_val, in_val, 1);
        
        // 최댓값 갱신
        if (current_wishes > max_wishes) {
            max_wishes = current_wishes;
        }
    }
    
    cout << max_wishes << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
